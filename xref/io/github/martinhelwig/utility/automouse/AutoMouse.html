5   package io.github.martinhelwig.utility.automouse;
6   
7   import java.awt.AWTException;
8   import java.awt.SystemTray;
9   import java.io.File;
10  import java.util.Arrays;
11  import java.util.Iterator;
12  import java.util.List;
13  import java.util.ResourceBundle;
14  
15  import org.apache.commons.logging.Log;
16  import org.apache.commons.logging.LogFactory;
17  
18  import io.github.martinhelwig.utility.automouse.common.ExcludeFromCodeCoverage;
19  
20  /**
21   * AutoMouse is the main class to start the program.
22   *
23   * @author Martin Helwig
24   *
25   */
26  @ExcludeFromCodeCoverage
27  public class AutoMouse {
28  
29      /**
30       * The logger.
31       *
32       * @since 1.0.0
33       */
34      private static Log logger = LogFactory.getLog(AutoMouse.class);
35  
36      /**
37       * The resource bundle.
38       *
39       * @since 1.0.0
40       */
41      private static ResourceBundle resourceBundle = ResourceBundle
42              .getBundle("io.github.martinhelwig.utility.automouse.languages."
43                      + "ApplicationResources");
44  
45      /**
46       * Default constructor.
47       *
48       * @since 1.0.0
49       */
50      public AutoMouse() {
51          logger.trace("AutoMouse() - start"); //$NON-NLS-1$
52          logger.trace("AutoMouse() - end"); //$NON-NLS-1$
53      }
54  
55      /**
56       * Logging libraries.
57       */
58      private void logLibraries() {
59          logger.trace("logLibraries() - start"); //$NON-NLS-1$
60  
61          List<String> classpath = Arrays.asList(System
62                  .getProperty("java.class.path").split(File.pathSeparator));
63  
64          Iterator<String> classpathIterator = classpath.iterator();
65          while (classpathIterator.hasNext()) {
66              String classpathEntry = classpathIterator.next();
67              logger.info(classpathEntry);
68          }
69  
70          logger.trace("logLibraries() - end"); //$NON-NLS-1$
71      }
72  
73      /**
74       * Running the auto-mouse.
75       *
76       * @throws AWTException
77       */
78      private void runAutoMouse() throws AWTException {
79          logger.trace("runAutoMouse() - start"); //$NON-NLS-1$
80  
81          AutoMouseInterface autoMouse = null;
82          if (SystemTray.isSupported()) {
83              autoMouse = new AutoMouseTrayIcon();
84          } else {
85              autoMouse = new AutoMouseCommon();
86          }
87  
88          try {
89              autoMouse.run();
90          } catch (Exception e) {
91              logger.error("an error occured, closing the program", e);
92              System.exit(1);
93          }
94  
95          logger.trace("runAutoMouse() - end"); //$NON-NLS-1$
96      }
97  
98      /**
99       * Main.
100      * @param args program arguments
101      */
102     public static void main(final String[] args) {
103         logger.info(resourceBundle.getString("programName") + " started...");
104         try {
105             AutoMouse mainProgram = new AutoMouse();
106             mainProgram.logLibraries();
107             mainProgram.runAutoMouse();
108         } catch (Exception e) {
109             logger.error(e);
110             e.printStackTrace();
111         }
112     }
113 }
