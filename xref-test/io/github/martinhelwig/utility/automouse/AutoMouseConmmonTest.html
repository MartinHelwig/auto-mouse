5   package io.github.martinhelwig.utility.automouse;
6   
7   import static org.junit.jupiter.api.Assertions.assertNotNull;
8   import static org.junit.jupiter.api.Assertions.fail;
9   
10  import java.awt.AWTException;
11  import java.awt.TrayIcon.MessageType;
12  import java.io.IOException;
13  import java.util.concurrent.ExecutorService;
14  import java.util.concurrent.Executors;
15  import java.util.concurrent.TimeUnit;
16  
17  import org.junit.jupiter.api.BeforeEach;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.TestMethodOrder;
20  
21  import io.github.martinhelwig.utility.automouse.common.Constants;
22  
23  import org.junit.jupiter.api.MethodOrderer.OrderAnnotation;
24  import org.junit.jupiter.api.Order;
25  
26  @TestMethodOrder(OrderAnnotation.class)
27  public class AutoMouseConmmonTest {
28  
29      /**
30       * The auto-mouse interface.
31       */
32      private AutoMouseInterface autoMouse = null;
33  
34      /**
35       * Setup.
36       */
37      @BeforeEach
38      public void setUp() {
39          autoMouse = new AutoMouseCommon();
40      }
41  
42      /**
43       * Test 1.
44       */
45      @Test @Order(Constants.ONE)
46      protected void testNotNull() {
47          try {
48              assertNotNull(autoMouse.moveMouse());
49              assertNotNull(autoMouse.getConfiguration());
50              assertNotNull(autoMouse.getLastLocation());
51              assertNotNull(autoMouse.isScreenLocked());
52              autoMouse.setCofiguration(new AutoMouseConfiguration());
53          } catch (AWTException | IOException e) {
54              fail(e);
55          }
56      }
57  
58      /**
59       * Test 2.
60       */
61      @Test @Order(Constants.TWO)
62      protected void testPushMessage() {
63          try {
64              autoMouse.pushMessage("JUnit execution - ERROR", MessageType.ERROR);
65              autoMouse.pushMessage("JUnit execution - WARNING",
66                      MessageType.WARNING);
67              autoMouse.pushMessage("JUnit execution - INFO", MessageType.INFO);
68              autoMouse.pushMessage("JUnit execution - NONE", MessageType.NONE);
69          } catch (Exception e) {
70              fail(e);
71          }
72      }
73  
74      /**
75       * Test 3.
76       */
77      @Test @Order(Constants.THREE)
78      protected void testRunnable() {
79          try {
80              autoMouse.getConfiguration().setTimer(1);
81              ExecutorService service = Executors.newSingleThreadExecutor();
82              service.execute(autoMouse);
83              service.shutdown();
84              service.awaitTermination(Constants.THREE, TimeUnit.SECONDS);
85          } catch (Exception e) {
86              fail(e);
87          }
88      }
89  }
